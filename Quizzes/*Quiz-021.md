## Code

```py
from matplotlib import pyplot as plt
import random

def produce(n, m, s):
    random.seed(1234)
    y = []
    x = []
    for i in range(n):
        x_rnd = random.randint(1,100)
        y_rnd = x_rnd**(-0.5*((m/s)**2))
        y.append(y_rnd)
        x.append(x_rnd)
    return x, y


y, x = produce(n=5, m=3, s=2)
print(y, x)
plt.plot(x, y)
plt.show()
```
